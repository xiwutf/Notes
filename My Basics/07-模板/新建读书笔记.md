<%*
const title = await tp.system.prompt("书名（如：思考快与慢）");
if (!title || !title.trim()) { tR = ""; return; }
const fileName = title.trim().startsWith("BOOK-") ? title.trim() : "BOOK-" + title.trim();
const folderPath = "03-知识库/读书";
const filePath = folderPath + "/" + fileName + ".md";
const existing = app.vault.getAbstractFileByPath(filePath);
if (existing) {
  await app.workspace.getLeaf(true).openFile(existing);
  new Notice("已打开：" + fileName);
  tR = "";
  return;
}
const template = tp.file.find_tfile("07-模板/读书笔记模板.md");
const folder = app.vault.getAbstractFileByPath(folderPath);
if (!template || !folder) {
  new Notice("模板或文件夹不存在，请检查配置");
  tR = "";
  return;
}
await tp.file.create_new(template, fileName, true, folder);
new Notice("已创建：" + fileName);
tR = "";
%>
